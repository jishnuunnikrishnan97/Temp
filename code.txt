from dateutil.parser import parse

def extract_date(text):
    try:
        date_obj = parse(text, fuzzy=True)
        return date_obj.strftime('%Y-%m-%d %H:%M:%S')  # Format the date as you want
    except ValueError:
        return None

# Example usage:
date_string = "March 28, 2023"
result = extract_date(date_string)
if result:
    print(result)
else:
    print("No valid date found in the text.")
