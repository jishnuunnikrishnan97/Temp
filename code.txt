import os
import shutil
from datetime import datetime

# Create a timestamp in the format "Prev Run YYYYMMDD-HHMM"
timestamp = datetime.now().strftime("Prev Run %Y%m%d-%H%M")

# Set the names of the source and destination directories
source_directory = "error pdf"  # Replace with your current error pdf folder name
archive_directory = "archive"

# Create the "archive" directory if it doesn't exist
if not os.path.exists(archive_directory):
    os.mkdir(archive_directory)

# Rename and move the previous "error pdf" folder to the "archive" directory
shutil.move(source_directory, os.path.join(archive_directory, timestamp))
