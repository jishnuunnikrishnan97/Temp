import pandas as pd

# Sample lists with different lengths
list1 = [1, 2, 3, 4]
list2 = ['A', 'B', 'C']
list3 = [10, 20]

# Find the maximum length among the lists
max_length = max(len(list1), len(list2), len(list3))

# Create a DataFrame with padded lists
df = pd.DataFrame({'Column1': list1 + [None] * (max_length - len(list1),
                  'Column2': list2 + [None] * (max_length - len(list2)),
                  'Column3': list3 + [None] * (max_length - len(list3))})

print(df)
