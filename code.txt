import pandas as pd
from PyPDF2 import PdfReader

# Load data
data = pd.read_excel("customers.xlsx")
customer_ids = data["customer_id"]
customer_names = data["customer_name"]

# Load PDF
reader = PdfReader("document.pdf")

# Extract text
page_text = []
for page in range(reader.getNumPages()):
    page_text.append(reader.getPage(page).extractText())

# Search and store matches
matches = {}
for name, customer_id in zip(customer_names, customer_ids):
    for page_number, page in enumerate(page_text):
        # Search using str.find or str.index
        index = page.find(name)
        if index != -1:
            matches[name] = (customer_id, page_number)
