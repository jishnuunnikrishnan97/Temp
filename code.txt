main_string = "This is a string reaffirming the rating of cheeta brain; child to example."
start_marker = "reaffirming the rating of"
end_marker = "to"

start_index = main_string.find(start_marker)
end_index = main_string.find(end_marker, start_index + len(start_marker))

if start_index != -1 and end_index != -1:
    desired_substring = main_string[start_index + len(start_marker):end_index].strip()
    print("Desired substring:", desired_substring)
else:
    print("Markers not found in the string.")
