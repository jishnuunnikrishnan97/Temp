import fitz  # PyMuPDF

def find_table_page(pdf_path, keyword):
    doc = fitz.open(pdf_path)
    page_number = None

    for page_num in range(doc.page_count):
        page = doc.load_page(page_num)
        text = page.get_text()

        if keyword in text:
            page_number = page_num + 1  # Page numbers are 1-based

    doc.close()
    return page_number

# Usage example:
pdf_path = 'your_pdf_file.pdf'
keyword = 'Table 10'
page_number = find_table_page(pdf_path, keyword)

if page_number is not None:
    print(f"'{keyword}' found on page {page_number}")
else:
    print(f"'{keyword}' not found in the PDF.")
