import pandas as pd

# Your lists
not_a_pdf = ["SUDARSHAN CHEMICAL INDUSTRIES LIMITED.pdf"]
corrupted = ['AngelOnelimited_February 07, 2023_RR_310943.pdf', 'Bajaj Finance Limited_April 18, 2023_RR_317703.pdf']
image_pdf = ['Fullerton India Credit Company Limited_ICRA (2).pdf', 'L&T Finance ICRA_LTF_BLR_19 12 2022.pdf']
independent = ['CEAT LIMITED.pdf']
no_HSBC_data = ["Decor Home Fabrics Private Limited.pdf", 'JSW Cement Limited.pdf']

# Create a list of tuples containing filename and remark using list comprehensions
data_tuples = [(filename, 'Not a pdf') for filename in not_a_pdf] + \
              [(filename, 'Corrupted') for filename in corrupted] + \
              [(filename, 'Image pdf') for filename in image_pdf] + \
              [(filename, 'Independent') for filename in independent] + \
              [(filename, 'no HSBC data') for filename in no_HSBC_data]

# Create a dataframe from the list of tuples
df = pd.DataFrame(data_tuples, columns=['Filename', 'remark'])

# Modify the "Filename" column as per your requirements
df['Filename'] = df['Filename'].apply(lambda x: x.replace('_', ' ').capitalize())

# Display the resulting dataframe
print(df)
