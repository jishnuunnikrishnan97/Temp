import re

# Sample data
data = {
    "Facilities": ["Gym", "Pool", "Tennis Court"],
}

kin = ["Tennis Court", "Gym", "Sauna", "Pool"]

# Create a list of strings from the 'Facilities' column
facility_list = [facility.lower() for facility in data["Facilities"]]

# Initialize an empty list to store the split 'kin' list
split_kin = []

# Iterate through the 'kin' list
for item in kin:
    temp_list = [item]
    # Iterate through the 'facility_list'
    for facility in facility_list:
        # Use regular expressions to check for substrings ignoring case
        if re.search(re.escape(facility), item, re.I):
            # Split the 'kin' item and add to 'temp_list'
            temp_list.extend(re.split(f'({re.escape(facility)})', item, flags=re.I))
    split_kin.extend(temp_list)

# Remove empty strings and None values
split_kin = [item for item in split_kin if item and item != ""]

print(split_kin)
