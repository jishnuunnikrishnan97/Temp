import re

def extract_dates(text):
    # Define regular expressions for the date formats
    date_pattern_long = r"(January|February|March|April|May|June|July|August|September|October|November|December)\s+\d{1,2},\s+\d{4}"
    date_pattern_short = r"\d{1,2}\s+(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\s+\d{4}"
    year_month_pattern = r"(January|February|March|April|May|June|July|August|September|October|November|December)\s+\d{4}"

    # Search for the date patterns in the text
    long_date_match = re.search(date_pattern_long, text, re.IGNORECASE)
    short_date_match = re.search(date_pattern_short, text, re.IGNORECASE)
    year_month_match = re.search(year_month_pattern, text, re.IGNORECASE)

    if long_date_match:
        return long_date_match.group()
    elif short_date_match:
        return short_date_match.group()
    elif year_month_match:
        return year_month_match.group()
    else:
        return None

# Test the function with your examples
text = "he waited in line at the night of March 28, 2023 and couldnt comprehend the fact that he was alive"
text1 = "he waited in line at the night of December 2022 and couldnt comprehend the fact that he was alive"

result_text = extract_dates(text)
result_text1 = extract_dates(text1)

print(result_text)  # Output: "March 28, 2023"
print(result_text1)  # Output: "December 2022"
