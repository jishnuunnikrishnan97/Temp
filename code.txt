import re
from dateutil import parser

def extract_dates(text):
    date_formats = [
        r"%B %d, %Y",
        r"%d %B %Y",
        r"%B %Y"
    ]
    
    for date_format in date_formats:
        pattern = re.compile(r"\b\d{1,2} (?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec) \d{4}\b", re.IGNORECASE)
        match = pattern.search(text)
        if match:
            date_str = match.group()
            date = parser.parse(date_str)
            return date.strftime("%B %d, %Y")
    
    return None

text = "he waited in line at the night of March 28, 2023 and couldn't comprehend the fact that he was alive"
result = extract_dates(text)
print(result)
