import pandas as pd
import numpy as np

# Sample DataFrame with spaces
data = {'A': [' ', '  ', '123', ' 45 ', '  ', '678'], 'B': ['abc', ' ', 'def', ' ', '  ', 'ghi']}
df = pd.DataFrame(data)

# Define a function to replace spaces with NaN
def replace_space_with_nan(cell):
    return np.nan if cell.strip() == '' else cell

# Apply the function to each cell in the DataFrame
df = df.applymap(replace_space_with_nan)

print(df)
