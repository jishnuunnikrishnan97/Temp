from dateutil import parser

def extract_and_convert_date(text):
    try:
        parsed_date = parser.parse(text, fuzzy=True)
        return parsed_date
    except ValueError:
        return None

# Example usage:
date_strings = ["March 28, 2023", "21 Apr 2023", "December 2022", "This is some random text"]
for date_str in date_strings:
    date = extract_and_convert_date(date_str)
    if date is not None:
        print(f"Extracted Date: {date}")
    else:
        print("No date found in the text.")
